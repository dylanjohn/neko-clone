import { createSignal } from 'solid-js'
 
export default function Counter() {
  const [count, setCount] = createSignal(0)
 
  const add = () => setCount((i) => i + 1)
 
  return (
    <div class='flex gap-x-4'>
      <button type='button' onClick={add}>
        the count is {count()}
      </button>
    </div>
  )
}
